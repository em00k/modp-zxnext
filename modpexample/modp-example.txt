#program modp-example#autostart  10 ; simple example of .modp  20 ; setting reg 127 prevents the modp splash info  30 CLS   40 PRINT ".modp BASIC example"  50 PRINT '".modp[layer] lets you play Amigatracker modules from NextBasic,  does not require a pi0."''  60 PRINT '"Press any key start.": PAUSE 0 : CLS   70 PRINT ".modp feel.mod"'  80 .modp modules/feel.mod  90 PAUSE 10 100 PROC br() 110 CLS  120 PRINT "We can suppress any text by     setting REG 127<>255"'': REG 127,1 130 .$ modp m$ 140 PRINT "End of example.": PAUSE 50: PAUSE 0  150 CLS : PROC br() 160 GO TO %150 170 ;  180 DEFPROC br() 190   REG 127,255 200   PRINT '"Press any key to Browse": PAUSE 0 : CLS  210   .browse -p "Pick a .MOD" -t "MOD" m$ 220   PRINT m$ 230 .$ modp m$ 235 PAUSE 50 240 ENDPROC  250 ;  260 STOP  270 REMOUNT : SAVE "modp-example.bas" LINE %10
